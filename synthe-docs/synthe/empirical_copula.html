 13class EmpiricalCopula:
 14    """
 15    Empirical Copula implementation for multivariate dependence modeling.
 16
 17    This class implements a non-parametric copula based on the empirical distribution
 18    of the data. It can fit to multivariate data and generate samples that preserve
 19    the original dependence structure.
 20
 21    The empirical copula is defined as:
 22    C_n(u1, ..., ud) = (1/n) * sum(I(U1i <= u1, ..., Udi <= ud))
 23
 24    where U_ji are the pseudo-observations (ranks) of the original data.
 25    """
 26
 27    def __init__(
 28        self,
 29        smoothing_method: str = "none",
 30        jitter_scale: float = 0.01,
 31        boundary_correction: bool = True,
 32    ):
 33        """
 34        Initialize the Empirical Copula.
 35
 36        Parameters:
 37        -----------
 38        smoothing_method : str, default "none"
 39            Smoothing method for the empirical copula:
 40            - "none": Pure empirical copula (no smoothing)
 41            - "jitter": Add small random noise to avoid ties
 42        jitter_scale : float, default 0.0
 43            Scale of uniform jitter to add to pseudo-observations (0 = no jitter).
 44        boundary_correction : bool, default True
 45            Whether to apply boundary correction for kernel methods.
 46        """
 47        self.smoothing_method = smoothing_method
 48        self.jitter_scale = jitter_scale
 49        self.boundary_correction = boundary_correction
 50        # Fitted attributes
 51        self.is_fitted_ = False
 52        self.n_samples_ = None
 53        self.n_vars_ = None
 54        self.pseudo_observations_ = None
 55        self.original_data_ = None
 56        self.marginal_cdfs_ = []
 57        self.marginal_quantiles_ = []
 58        # For kernel-based methods
 59        self.kde_model_ = None
 60        # For Gaussian mixture model
 61        self.gmm_model_ = None

 63    def fit(self, X: np.ndarray) -> "EmpiricalCopula":
 64        """
 65        Fit the empirical copula to the data.
 66
 67        Parameters:
 68        -----------
 69        X : np.ndarray
 70            Input data of shape (n_samples, n_features) on original scale.
 71
 72        Returns:
 73        --------
 74        self : EmpiricalCopula
 75            Returns self for method chaining.
 76
 77        Raises:
 78        -------
 79        ValueError
 80            If X has inappropriate dimensions.
 81        """
 82        X = np.asarray(X)
 83
 84        if X.ndim != 2:
 85            raise ValueError("X must be a 2D array")
 86        if X.shape[1] < 2:
 87            raise ValueError("X must have at least 2 variables")
 88        if X.shape[0] < 2:
 89            raise ValueError("X must have at least 2 observations")
 90
 91        self.n_samples_, self.n_vars_ = X.shape
 92        self.original_data_ = X.copy()
 93        # Step 1: Convert to pseudo-observations (ranks)
 94        self.pseudo_observations_ = self._to_pseudo_observations(X)
 95        # Step 2: Apply smoothing if requested
 96        if self.smoothing_method != "none":
 97            self.pseudo_observations_ = self._apply_smoothing(
 98                self.pseudo_observations_
 99            )
100        # Step 3: Store marginal information for inverse transformation
101        self._fit_marginal_transforms(X)
102        self.is_fitted_ = True
103        # print(f"Empirical copula fitted successfully using '{self.smoothing_method}' method")
104        return self

106    def sample(
107        self,
108        n_samples: int = 50,
109        method: str = "bootstrap",
110        kernel: str = "gaussian",
111        n_components: int = 5,
112        covariance_type: str = "full",
113        return_pseudo: bool = False,
114        random_state: Optional[int] = 123,
115        **kwargs,
116    ) -> np.ndarray:
117        """
118        Generate samples from the fitted empirical copula.
119
120        Parameters:
121        -----------
122        n_samples : int, default 50
123            Number of samples to generate.
124        method : str, default "bootstrap"
125            Sampling method:
126            - "bootstrap": Bootstrap resampling from fitted pseudo-observations
127            - "kde": Kernel density estimation sampling (if smoothing was used)
128            - "gmm": Gaussian mixture model sampling
129        kernel : str, default "gaussian"
130            Kernel to use if method is "kde" (default is 'gaussian').
131            Can also be 'tophat'.
132        n_components : int, default 5
133            Number of Gaussian components for GMM method.
134        covariance_type : str, default "full"
135            Type of covariance parameters for GMM method.
136            Options: 'full', 'tied', 'diag', 'spherical'.
137        return_pseudo : bool, default False
138            If True, return samples on [0,1] copula scale.
139            If False, return samples transformed to original scale.
140        random_state : int, optional
141            Random state for reproducible sampling.
142        kwargs : additional arguments for specific sampling methods.
143
144        Returns:
145        --------
146        samples : np.ndarray
147            Generated samples of shape (n_samples, n_features).
148        """
149        if not self.is_fitted_:
150            raise ValueError(
151                "Copula must be fitted before sampling. Call fit() first."
152            )
153
154        if random_state is not None:
155            np.random.seed(random_state)
156        # Generate pseudo-observations
157        if method == "bootstrap":
158            pseudo_samples = self._bootstrap_sample(n_samples)
159        elif method == "kde":
160            pseudo_samples = self._kde_sample(
161                n_samples, kernel=kernel, **kwargs
162            )
163        elif method == "gmm":
164            pseudo_samples = self._gmm_sample(
165                n_samples,
166                n_components=n_components,
167                covariance_type=covariance_type,
168                **kwargs,
169            )
170        else:
171            raise ValueError(
172                f"Unknown sampling method: {method}. "
173                f"Supported methods: 'bootstrap', 'kde', 'gmm'"
174            )
175        if return_pseudo:
176            return pseudo_samples
177        # Transform back to original scale
178        return self._inverse_transform(pseudo_samples)

205    def estimate_tail_dependence(self, threshold=0.05):
206        tail_dep = {}
207        for i in range(self.n_vars_):
208            for j in range(i + 1, self.n_vars_):
209                u = self.pseudo_observations_[:, i]
210                v = self.pseudo_observations_[:, j]
211                lower_tail = (
212                    np.mean((u < threshold) & (v < threshold)) / threshold
213                )
214                upper_tail = (
215                    np.mean((u > 1 - threshold) & (v > 1 - threshold))
216                    / threshold
217                )
218                tail_dep[f"var{i+1}-var{j+1}"] = {
219                    "lower": lower_tail,
220                    "upper": upper_tail,
221                }
222        return tail_dep

249    def validate_fit(
250        self,
251        X_test: Optional[np.ndarray] = None,
252        n_bootstrap: int = 250,
253        alpha: float = 0.05,
254        verbose: bool = True,
255    ) -> Dict:
256        """
257        Validate the fitted empirical copula using comprehensive hypothesis tests.
258
259        This method performs:
260        1. Kolmogorov-Smirnov tests on marginal distributions
261        2. Anderson-Darling tests for marginal goodness-of-fit
262        3. Tests for dependence measures (Spearman rho, Kendall tau, Pearson correlation)
263        4. Cramér-von Mises test for copula goodness-of-fit
264        5. Tests for uniform distribution of pseudo-observations
265
266        Parameters:
267        -----------
268        X_test : np.ndarray, optional
269            Test data for validation. If None, uses training data.
270        n_bootstrap : int, default 1000
271            Number of bootstrap samples for validation.
272        alpha : float, default 0.05
273            Significance level for statistical tests.
274        verbose : bool, default True
275            Whether to print detailed validation results.
276
277        Returns:
278        --------
279        validation_results : dict
280            Dictionary containing all validation test results.
281        """
282        if not self.is_fitted_:
283            raise ValueError("Copula must be fitted before validation.")
284
285        # Use training data if no test data provided
286        if X_test is None:
287            X_test = self.original_data_.copy()
288            if verbose:
289                print("Note: Using training data for validation")
290
291        # Generate bootstrap samples for comparison
292        bootstrap_samples = self.sample(n_samples=n_bootstrap, random_state=42)
293
294        results = {
295            "marginal_tests": {},
296            "dependence_tests": {},
297            "copula_tests": {},
298            "uniformity_tests": {},
299            "summary": {},
300        }
301
302        if verbose:
303            print("\n=== EMPIRICAL COPULA VALIDATION TESTS ===\n")
304
305        # 1. MARGINAL DISTRIBUTION TESTS
306        if verbose:
307            print("1. Marginal Distribution Tests:")
308            print("-" * 35)
309
310        for j in range(self.n_vars_):
311            original_margin = X_test[:, j]
312            simulated_margin = bootstrap_samples[:, j]
313            # Kolmogorov-Smirnov test
314            ks_stat, ks_pvalue = stats.ks_2samp(
315                original_margin, simulated_margin
316            )
317            # Anderson-Darling test (if samples are from same distribution)
318            try:
319                # Combine samples and test if they're from the same distribution
320                combined_data = np.concatenate(
321                    [original_margin, simulated_margin]
322                )
323                combined_mean = np.mean(combined_data)
324                combined_std = np.std(combined_data)
325                # Test both against normal distribution with combined parameters
326                ad_orig = stats.anderson(original_margin, dist="norm")
327                ad_sim = stats.anderson(simulated_margin, dist="norm")
328                # Use the test statistic difference as a measure
329                ad_diff = abs(ad_orig.statistic - ad_sim.statistic)
330                ad_critical = ad_orig.critical_values[
331                    2
332                ]  # 5% significance level
333                ad_pass = ad_diff < ad_critical * 0.5  # Heuristic threshold
334            except:
335                ad_diff = np.nan
336                ad_pass = None
337            # Two-sample t-test for means
338            ttest_stat, ttest_pvalue = stats.ttest_ind(
339                original_margin, simulated_margin
340            )
341            # Levene's test for equal variances
342            levene_stat, levene_pvalue = stats.levene(
343                original_margin, simulated_margin
344            )
345
346            results["marginal_tests"][f"variable_{j+1}"] = {
347                "ks_statistic": ks_stat,
348                "ks_p_value": ks_pvalue,
349                "ks_reject_null": ks_pvalue < alpha,
350                "ad_difference": ad_diff,
351                "ad_pass": ad_pass,
352                "ttest_statistic": ttest_stat,
353                "ttest_p_value": ttest_pvalue,
354                "mean_difference_significant": ttest_pvalue < alpha,
355                "levene_statistic": levene_stat,
356                "levene_p_value": levene_pvalue,
357                "variance_difference_significant": levene_pvalue < alpha,
358            }
359
360            if verbose:
361                status = "FAIL" if ks_pvalue < alpha else "PASS"
362                mean_status = "FAIL" if ttest_pvalue < alpha else "PASS"
363                var_status = "FAIL" if levene_pvalue < alpha else "PASS"
364                print(f"Variable {j+1}:")
365                print(
366                    f"  KS test: statistic={ks_stat:.4f}, p-value={ks_pvalue:.4f} [{status}]"
367                )
368                print(
369                    f"  Mean test: p-value={ttest_pvalue:.4f} [{mean_status}]"
370                )
371                print(
372                    f"  Variance test: p-value={levene_pvalue:.4f} [{var_status}]"
373                )
374
375        # 2. DEPENDENCE STRUCTURE TESTS
376        if verbose:
377            print(f"\n2. Dependence Structure Tests:")
378            print("-" * 32)
379
380        # Calculate dependence measures
381        orig_corr = np.corrcoef(X_test.T)
382        orig_spearman = self._calculate_spearman_matrix(X_test)
383        orig_kendall = self._calculate_kendall_matrix(X_test)
384
385        sim_corr = np.corrcoef(bootstrap_samples.T)
386        sim_spearman = self._calculate_spearman_matrix(bootstrap_samples)
387        sim_kendall = self._calculate_kendall_matrix(bootstrap_samples)
388
389        # Statistical tests for dependence measures
390        dependence_results = {}
391
392        for i in range(self.n_vars_):
393            for j in range(i + 1, self.n_vars_):
394                pair_name = f"var{i+1}_var{j+1}"
395                # Test correlations using Fisher's z-transform
396                r1, r2 = orig_corr[i, j], sim_corr[i, j]
397                n1, n2 = len(X_test), len(bootstrap_samples)
398                # Fisher's z-transform
399                z1 = (
400                    0.5 * np.log((1 + r1) / (1 - r1))
401                    if abs(r1) < 0.999
402                    else np.sign(r1) * 3
403                )
404                z2 = (
405                    0.5 * np.log((1 + r2) / (1 - r2))
406                    if abs(r2) < 0.999
407                    else np.sign(r2) * 3
408                )
409                # Test statistic
410                se = np.sqrt(1 / (n1 - 3) + 1 / (n2 - 3))
411                z_stat = (z1 - z2) / se if se > 0 else 0
412                corr_pvalue = 2 * (1 - stats.norm.cdf(abs(z_stat)))
413                # Spearman and Kendall differences
414                spear_diff = abs(orig_spearman[i, j] - sim_spearman[i, j])
415                kendall_diff = abs(orig_kendall[i, j] - sim_kendall[i, j])
416
417                dependence_results[pair_name] = {
418                    "pearson_original": r1,
419                    "pearson_simulated": r2,
420                    "pearson_z_statistic": z_stat,
421                    "pearson_p_value": corr_pvalue,
422                    "pearson_significant_diff": corr_pvalue < alpha,
423                    "spearman_difference": spear_diff,
424                    "kendall_difference": kendall_diff,
425                    "spearman_large_diff": spear_diff > 0.1,
426                    "kendall_large_diff": kendall_diff > 0.1,
427                }
428
429        results["dependence_tests"] = dependence_results
430
431        if verbose:
432            for pair, tests in dependence_results.items():
433                corr_status = (
434                    "FAIL" if tests["pearson_significant_diff"] else "PASS"
435                )
436                spear_status = (
437                    "WARN" if tests["spearman_large_diff"] else "PASS"
438                )
439                kendall_status = (
440                    "WARN" if tests["kendall_large_diff"] else "PASS"
441                )
442                print(f"{pair.replace('_', '-')}:")
443                print(
444                    f"  Pearson: {tests['pearson_original']:.4f} vs {tests['pearson_simulated']:.4f}, "
445                    f"p-val={tests['pearson_p_value']:.4f} [{corr_status}]"
446                )
447                print(
448                    f"  Spearman diff: {tests['spearman_difference']:.4f} [{spear_status}]"
449                )
450                print(
451                    f"  Kendall diff: {tests['kendall_difference']:.4f} [{kendall_status}]"
452                )
453        # 3. UNIFORMITY TESTS FOR PSEUDO-OBSERVATIONS
454        if verbose:
455            print(f"\n3. Uniformity Tests (Pseudo-Observations):")
456            print("-" * 42)
457        # Test if pseudo-observations are uniform on [0,1]
458        pseudo_test = self._to_pseudo_observations(X_test)
459
460        uniformity_results = {}
461
462        for j in range(self.n_vars_):
463            pseudo_margin = pseudo_test[:, j]
464            # Kolmogorov-Smirnov test against uniform distribution
465            ks_uniform_stat, ks_uniform_pvalue = stats.kstest(
466                pseudo_margin, "uniform"
467            )
468            # Anderson-Darling test for uniformity
469            # Transform to standard normal and test
470            normal_transformed = stats.norm.ppf(
471                np.clip(pseudo_margin, 1e-10, 1 - 1e-10)
472            )
473            ad_result = stats.anderson(normal_transformed, dist="norm")
474
475            # Cramer-von Mises test for uniformity
476            def cvm_uniform(data):
477                """Cramér-von Mises test for uniform distribution."""
478                n = len(data)
479                sorted_data = np.sort(data)
480                i = np.arange(1, n + 1)
481                T = (1.0 / (12 * n)) + np.sum(
482                    ((2 * i - 1) / (2 * n) - sorted_data) ** 2
483                )
484                return T
485
486            cvm_stat = cvm_uniform(pseudo_margin)
487            # Critical value at 5% significance level
488            cvm_critical = 0.461 / (
489                np.sqrt(len(pseudo_margin))
490                + 0.25
491                + 0.75 / np.sqrt(len(pseudo_margin))
492            )
493
494            uniformity_results[f"variable_{j+1}"] = {
495                "ks_uniform_statistic": ks_uniform_stat,
496                "ks_uniform_p_value": ks_uniform_pvalue,
497                "ks_uniform_reject": ks_uniform_pvalue < alpha,
498                "ad_statistic": ad_result.statistic,
499                "ad_critical_5pct": ad_result.critical_values[2],
500                "ad_reject": ad_result.statistic > ad_result.critical_values[2],
501                "cvm_statistic": cvm_stat,
502                "cvm_critical": cvm_critical,
503                "cvm_reject": cvm_stat > cvm_critical,
504            }
505
506            if verbose:
507                ks_status = "FAIL" if ks_uniform_pvalue < alpha else "PASS"
508                ad_status = (
509                    "FAIL"
510                    if ad_result.statistic > ad_result.critical_values[2]
511                    else "PASS"
512                )
513                cvm_status = "FAIL" if cvm_stat > cvm_critical else "PASS"
514                print(f"Variable {j+1}:")
515                print(
516                    f"  KS uniform: stat={ks_uniform_stat:.4f}, p-val={ks_uniform_pvalue:.4f} [{ks_status}]"
517                )
518                print(
519                    f"  AD normal: stat={ad_result.statistic:.4f}, crit={ad_result.critical_values[2]:.4f} [{ad_status}]"
520                )
521                print(
522                    f"  CvM uniform: stat={cvm_stat:.4f}, crit={cvm_critical:.4f} [{cvm_status}]"
523                )
524
525        results["uniformity_tests"] = uniformity_results
526
527        pseudo_orig = self._to_pseudo_observations(X_test)
528        pseudo_sim = self._to_pseudo_observations(bootstrap_samples)
529        # 5. SUMMARY ASSESSMENT
530        if verbose:
531            print(f"\n5. Overall Assessment:")
532            print("-" * 22)
533        # Count various test failures
534        ks_failures = sum(
535            1
536            for j in range(self.n_vars_)
537            if results["marginal_tests"][f"variable_{j+1}"]["ks_reject_null"]
538        )
539
540        mean_failures = sum(
541            1
542            for j in range(self.n_vars_)
543            if results["marginal_tests"][f"variable_{j+1}"][
544                "mean_difference_significant"
545            ]
546        )
547
548        var_failures = sum(
549            1
550            for j in range(self.n_vars_)
551            if results["marginal_tests"][f"variable_{j+1}"][
552                "variance_difference_significant"
553            ]
554        )
555
556        corr_failures = sum(
557            1
558            for tests in dependence_results.values()
559            if tests["pearson_significant_diff"]
560        )
561
562        uniform_failures = sum(
563            1
564            for j in range(self.n_vars_)
565            if results["uniformity_tests"][f"variable_{j+1}"][
566                "ks_uniform_reject"
567            ]
568        )
569        # Calculate average differences
570        avg_spear_diff = np.mean(
571            [
572                tests["spearman_difference"]
573                for tests in dependence_results.values()
574            ]
575        )
576        avg_kendall_diff = np.mean(
577            [
578                tests["kendall_difference"]
579                for tests in dependence_results.values()
580            ]
581        )
582        # Overall quality assessment
583        total_tests = (
584            self.n_vars_ * 3 + len(dependence_results) + self.n_vars_ + 1
585        )
586        total_failures = (
587            ks_failures
588            + mean_failures
589            + var_failures
590            + corr_failures
591            + uniform_failures
592        )
593
594        pass_rate = (total_tests - total_failures) / total_tests * 100
595
596        if pass_rate >= 85 and avg_spear_diff <= 0.05:
597            quality = "Excellent"
598        elif pass_rate >= 70 and avg_spear_diff <= 0.10:
599            quality = "Good"
600        elif pass_rate >= 50 and avg_spear_diff <= 0.15:
601            quality = "Fair"
602        else:
603            quality = "Poor"
604
605        results["summary"] = {
606            "ks_failures": ks_failures,
607            "mean_failures": mean_failures,
608            "variance_failures": var_failures,
609            "correlation_failures": corr_failures,
610            "uniformity_failures": uniform_failures,
611            "total_failures": total_failures,
612            "total_tests": total_tests,
613            "pass_rate": pass_rate,
614            "avg_spearman_difference": avg_spear_diff,
615            "avg_kendall_difference": avg_kendall_diff,
616            "overall_quality": quality,
617        }
618
619        if verbose:
620            print(f"Test Summary ({total_tests} total tests):")
621            print(f"  Marginal KS failures: {ks_failures}/{self.n_vars_}")
622            print(f"  Mean difference failures: {mean_failures}/{self.n_vars_}")
623            print(
624                f"  Variance difference failures: {var_failures}/{self.n_vars_}"
625            )
626            print(
627                f"  Correlation failures: {corr_failures}/{len(dependence_results)}"
628            )
629            print(f"  Uniformity failures: {uniform_failures}/{self.n_vars_}")
630            print(f"  Overall pass rate: {pass_rate:.1f}%")
631            print(f"  Average Spearman difference: {avg_spear_diff:.4f}")
632            print(f"  Average Kendall difference: {avg_kendall_diff:.4f}")
633            print(f"  Overall model quality: {quality}")
634
635        return results

637    def _to_pseudo_observations(self, X: np.ndarray) -> np.ndarray:
638        """Convert data to pseudo-observations using empirical CDF."""
639        n_samples, n_vars = X.shape
640        pseudo_obs = np.zeros_like(X)
641
642        for j in range(n_vars):
643            # Rank-based transformation
644            ranks = stats.rankdata(X[:, j], method="average")
645            # Use (rank - 0.5) / n to avoid boundary values
646            pseudo_obs[:, j] = (ranks - 0.5) / n_samples
647
648        return pseudo_obs

650    def _apply_smoothing(self, pseudo_obs: np.ndarray) -> np.ndarray:
651        """Apply smoothing to pseudo-observations."""
652        if self.smoothing_method == "jitter":
653            # Add uniform jitter
654            jitter = np.random.uniform(
655                -self.jitter_scale / 2, self.jitter_scale / 2, pseudo_obs.shape
656            )
657            smoothed = pseudo_obs + jitter
658            # Ensure values stay in [0,1]
659            smoothed = np.clip(smoothed, 1e-10, 1 - 1e-10)
660            return smoothed
661        else:
662            return pseudo_obs

664    def _fit_marginal_transforms(self, X: np.ndarray) -> None:
665        """Fit marginal transformations for inverse sampling."""
666        self.marginal_cdfs_ = []
667        self.marginal_quantiles_ = []
668
669        for j in range(self.n_vars_):
670            data_col = X[:, j]
671            sorted_data = np.sort(data_col)
672            # Create empirical CDF
673            n = len(sorted_data)
674            cdf_values = np.arange(1, n + 1) / n
675            # Store quantile function (inverse CDF)
676            # Add boundary extrapolation
677            extended_probs = np.concatenate([[0], cdf_values, [1]])
678            extended_data = np.concatenate(
679                [
680                    [sorted_data[0] - (sorted_data[1] - sorted_data[0])],
681                    sorted_data,
682                    [sorted_data[-1] + (sorted_data[-1] - sorted_data[-2])],
683                ]
684            )
685            quantile_func = interp1d(
686                extended_probs,
687                extended_data,
688                kind="linear",
689                bounds_error=False,
690                fill_value=(extended_data[0], extended_data[-1]),
691            )
692            self.marginal_quantiles_.append(quantile_func)

694    def _bootstrap_sample(self, n_samples: int) -> np.ndarray:
695        """Generate samples using bootstrap resampling."""
696        # Randomly sample indices with replacement
697        indices = np.random.choice(
698            self.n_samples_, size=n_samples, replace=True
699        )
700        return self.pseudo_observations_[indices]

255        verbose: bool = True,
256    ) -> Dict:

258        Validate the fitted empirical copula using comprehensive hypothesis tests.
259
260        This method performs:
261        1. Kolmogorov-Smirnov tests on marginal distributions
262        2. Anderson-Darling tests for marginal goodness-of-fit
263        3. Tests for dependence measures (Spearman rho, Kendall tau, Pearson correlation)

307        if verbose:

316                original_margin, simulated_margin
317            )
318            # Anderson-Darling test (if samples are from same distribution)
319            try:
320                # Combine samples and test if they're from the same distribution
321                combined_data = np.concatenate(
322                    [original_margin, simulated_margin]

325                combined_std = np.std(combined_data)
326                # Test both against normal distribution with combined parameters
327                ad_orig = stats.anderson(original_margin, dist="norm")

330                ad_diff = abs(ad_orig.statistic - ad_sim.statistic)
331                ad_critical = ad_orig.critical_values[
332                    2
333                ]  # 5% significance level
334                ad_pass = ad_diff < ad_critical * 0.5  # Heuristic threshold
335            except:
336                ad_diff = np.nan

456            print(f"\n3. Uniformity Tests (Pseudo-Observations):")
457            print("-" * 42)

479                n = len(data)

638    def _to_pseudo_observations(self, X: np.ndarray) -> np.ndarray:
639        """Convert data to pseudo-observations using empirical CDF."""
640        n_samples, n_vars = X.shape
641        pseudo_obs = np.zeros_like(X)
642
643        for j in range(n_vars):
644            # Rank-based transformation
645            ranks = stats.rankdata(X[:, j], method="average")
646            # Use (rank - 0.5) / n to avoid boundary values
647            pseudo_obs[:, j] = (ranks - 0.5) / n_samples
648
649        return pseudo_obs

651    def _apply_smoothing(self, pseudo_obs: np.ndarray) -> np.ndarray:
652        """Apply smoothing to pseudo-observations."""
653        if self.smoothing_method == "jitter":
654            # Add uniform jitter
655            jitter = np.random.uniform(
656                -self.jitter_scale / 2, self.jitter_scale / 2, pseudo_obs.shape
657            )
658            smoothed = pseudo_obs + jitter
659            # Ensure values stay in [0,1]
660            smoothed = np.clip(smoothed, 1e-10, 1 - 1e-10)
661            return smoothed
662        else:
663            return pseudo_obs

665    def _fit_marginal_transforms(self, X: np.ndarray) -> None:
666        """Fit marginal transformations for inverse sampling."""
667        self.marginal_cdfs_ = []
668        self.marginal_quantiles_ = []
669
670        for j in range(self.n_vars_):
671            data_col = X[:, j]
672            sorted_data = np.sort(data_col)
673            # Create empirical CDF
674            n = len(sorted_data)
675            cdf_values = np.arange(1, n + 1) / n
676            # Store quantile function (inverse CDF)
677            # Add boundary extrapolation
678            extended_probs = np.concatenate([[0], cdf_values, [1]])
679            extended_data = np.concatenate(
680                [
681                    [sorted_data[0] - (sorted_data[1] - sorted_data[0])],
682                    sorted_data,
683                    [sorted_data[-1] + (sorted_data[-1] - sorted_data[-2])],
684                ]
685            )
686            quantile_func = interp1d(
687                extended_probs,
688                extended_data,
689                kind="linear",
690                bounds_error=False,
691                fill_value=(extended_data[0], extended_data[-1]),
692            )
693            self.marginal_quantiles_.append(quantile_func)

695    def _bootstrap_sample(self, n_samples: int) -> np.ndarray:
696        """Generate samples using bootstrap resampling."""
697        # Randomly sample indices with replacement
698        indices = np.random.choice(
699            self.n_samples_, size=n_samples, replace=True
700        )
701        return self.pseudo_observations_[indices]

Empirical Copula implementation for multivariate dependence modeling.

This class implements a non-parametric copula based on the empirical distribution of the data. It can fit to multivariate data and generate samples that preserve the original dependence structure.

The empirical copula is defined as: C_n(u1, ..., ud) = (1/n) * sum(I(U1i <= u1, ..., Udi <= ud))

where U_ji are the pseudo-observations (ranks) of the original data.

 135    def _setup_jax_backend(self):
 136        """Setup JAX backend for GPU/TPU acceleration."""
 137        if not JAX_AVAILABLE:
 138            raise ImportError("JAX is required for GPU/TPU backend")
 139
 140        # JIT compiled distance functions
 141        @jit
 142        def pairwise_sq_dists_jax(X1, X2):
 143            X1_sq = jnp.sum(X1**2, axis=1)[:, jnp.newaxis]
 144            X2_sq = jnp.sum(X2**2, axis=1)[jnp.newaxis, :]
 145            return X1_sq + X2_sq - 2 * X1 @ X2.T

 147        @jit
 148        def cdist_jax(X1, X2):
 149            return vmap(
 150                lambda x: vmap(lambda y: jnp.sqrt(jnp.sum((x - y) ** 2)))(X2)

 236    def _sample_residuals(self, num_samples):
 237        """Sample residuals using the chosen method."""
 238        if self.residuals is None:
 239            raise ValueError("No residuals available for sampling")
 240
 241        if self.residual_sampling == "bootstrap":
 242            # Original bootstrap method
 243            n = self.residuals.shape[0]
 244            idx = np.random.choice(n, num_samples, replace=True)
 245            return self.residuals[idx]

 247        elif self.residual_sampling == "kde":

 255            samples = self.kde_model_.sample(num_samples)

 262                    "GMM model not fitted. Call _fit_residual_sampler first."

 274    def _pairwise_sq_dists(self, X1, X2):
 275        """Compute pairwise squared Euclidean distances."""
 276        if self.backend in ["gpu", "tpu"] and JAX_AVAILABLE:
 277            X1_jax = jnp.array(X1)
 278            X2_jax = jnp.array(X2)
 279            result = self._pairwise_sq_dists_jax(X1_jax, X2_jax)
 280            return np.array(result)
 281        else:
 282            X1 = np.atleast_2d(X1)
 283            X2 = np.atleast_2d(X2)
 284            return (
 285                np.sum(X1**2, axis=1)[:, np.newaxis]
 286                + np.sum(X2**2, axis=1)[np.newaxis, :]
 287                - 2 * X1 @ X2.T
 288            )

 327    def _mmd(self, u, v, kernel_sigma=1):
 328        """Maximum Mean Discrepancy between two distributions."""
 329        if u.ndim == 1:
 330            u = u.reshape(-1, 1)
 331        if v.ndim == 1:
 332            v = v.reshape(-1, 1)
 333
 334        def kmat(A, B):
 335            return np.exp(
 336                -self._pairwise_sq_dists(A, B) / (2 * kernel_sigma**2)
 337            )
 338
 339        return (
 340            np.mean(kmat(u, u)) + np.mean(kmat(v, v)) - 2 * np.mean(kmat(u, v))
 341        )

 343    def _custom_energy_distance(self, u, v):
 344        """Energy distance between two distributions."""
 345        if u.ndim == 1:
 346            u = u.reshape(-1, 1)
 347        if v.ndim == 1:
 348            v = v.reshape(-1, 1)
 349
 350        n, d = u.shape
 351        m = v.shape[0]
 352
 353        if self.backend in ["gpu", "tpu"] and JAX_AVAILABLE:
 354            # JAX implementation
 355            u_jax = jnp.array(u)
 356            v_jax = jnp.array(v)
 357
 358            dist_xx = self._cdist_jax(u_jax, u_jax)
 359            dist_yy = self._cdist_jax(v_jax, v_jax)
 360            dist_xy = self._cdist_jax(u_jax, v_jax)
 361
 362            term1 = 2 * jnp.sum(dist_xy) / (n * m)
 363            term2 = jnp.sum(dist_xx) / (n * n)
 364            term3 = jnp.sum(dist_yy) / (m * m)
 365
 366            return float(term1 - term2 - term3)
 367        else:
 368            # NumPy implementation
 369            dist_xx = cdist(u, u, metric="euclidean")
 370            dist_yy = cdist(v, v, metric="euclidean")
 371            dist_xy = cdist(u, v, metric="euclidean")
 372
 373            term1 = 2 * np.sum(dist_xy) / (n * m)
 374            term2 = np.sum(dist_xx) / (n * n)
 375            term3 = np.sum(dist_yy) / (m * m)
 376            return term1 - term2 - term3

 401    def fit(
 402        self, Y, X_dist, n_train=None, metric="energy", n_trials=50, **kwargs
 403    ):
 404        """
 405        Fit the data generator to match the distribution of Y.
 406
 407        Parameters:
 408        -----------
 409        Y : array-like, shape (n_samples, n_features)
 410            Target multivariate data to emulate
 411        X_dist : callable
 412            Function that takes n_samples and returns input data X
 413        n_train : int, default=None
 414            Number of training samples (default: n_samples // 2)
 415        metric : str, default='energy'
 416            Distance metric for optimization ('energy', 'mmd', or 'wasserstein')
 417        n_trials : int, default=50
 418            Number of Optuna optimization trials
 419        **kwargs : dict
 420            Additional arguments for Optuna optimization
 421
 422        Returns:
 423        --------
 424        self : object
 425            Returns self
 426        """
 427        if Y.ndim == 1:
 428            Y = Y.reshape(-1, 1)
 429
 430        n, d = Y.shape
 431        self.n_features_ = d
 432
 433        # Determine whether to use RFF
 434        if self.use_rff == "auto":
 435            self.actual_use_rff_ = n >= self.force_rff_threshold
 436        else:
 437            self.actual_use_rff_ = self.use_rff
 438
 439        # Auto-enable RFF for large datasets with component determination
 440        if self.actual_use_rff_:
 441            self.actual_rff_components_ = self._determine_components(n, d)
 442            if self.use_rff == "auto":
 443                print(
 444                    f"Large dataset detected (n={n}). Auto-enabling {self.kernel_approximation.upper()} for scalability."
 445                )
 446
 447        if n_train is None:
 448            n_train = n // 2
 449
 450        # Store the input distribution function
 451        self.X_dist = X_dist
 452
 453        # Create stratified train-test split
 454        train_idx, test_idx = self._stratified_train_test_split(Y, n_train)
 455        Y_train = Y[train_idx]
 456        Y_test = Y[test_idx]
 457        X_train = X_dist(n_train)
 458
 459        print(
 460            f"Stratified split: {len(train_idx)} train, {len(test_idx)} test samples"
 461        )
 462        print(f"Residual sampling method: {self.residual_sampling}")
 463        print(
 464            f"Using {self.kernel_approximation.upper()}: {self.actual_use_rff_}"
 465        )
 466        if self.actual_use_rff_:
 467            print(f"Components: {self.actual_rff_components_}")
 468            print(f"Approximation method: {self.kernel_approximation}")
 469        if self.residual_sampling == "kde":
 470            print(f"KDE bandwidth: {self.kde_bandwidth}")
 471        elif self.residual_sampling == "gmm":
 472            print(f"GMM components: {self.gmm_components}")
 473
 474        def objective(trial):
 475            sigma = trial.suggest_float("sigma", 0.01, 10, log=True)
 476            lambd = trial.suggest_float("lambd", 1e-5, 1, log=True)
 477            gamma = 1 / (2 * sigma**2)
 478
 479            # Create model with current parameters
 480            model = self._create_model(gamma, lambd)
 481            model.fit(X_train, Y_train)
 482
 483            preds_train = model.predict(X_train)
 484            if preds_train.ndim == 1:
 485                preds_train = preds_train.reshape(-1, 1)
 486
 487            res = Y_train - preds_train
 488            Y_sim = self._generate_pseudo_with_model(model, res, len(Y_test))
 489
 490            if metric == "energy":
 491                dist_val = self._custom_energy_distance(Y_test, Y_sim)
 492            elif metric == "mmd":
 493                dist_val = self._mmd(Y_test, Y_sim)
 494            elif metric == "wasserstein" and d == 1:
 495                dist_val = stats.wasserstein_distance(
 496                    Y_test.flatten(), Y_sim.flatten()
 497                )
 498            else:
 499                raise ValueError("Invalid metric for dimension")
 500
 501            return dist_val
 502
 503        # Optimize hyperparameters
 504        study = optuna.create_study(direction="minimize")
 505        study.optimize(objective, n_trials=n_trials, **kwargs)
 506
 507        # Store best parameters and fit final model
 508        self.best_params_ = study.best_params
 509        self.best_score_ = study.best_value
 510
 511        sigma = self.best_params_["sigma"]
 512        lambd = self.best_params_["lambd"]
 513        gamma = 1 / (2 * sigma**2)
 514
 515        # Fit final model with best parameters
 516        self.model = self._create_model(gamma, lambd)
 517        self.model.fit(X_train, Y_train)
 518
 519        # Compute residuals
 520        preds_train = self.model.predict(X_train)
 521        if preds_train.ndim == 1:
 522            preds_train = preds_train.reshape(-1, 1)
 523        self.residuals = Y_train - preds_train
 524
 525        # Fit the residual sampler
 526        self._fit_residual_sampler()
 527
 528        self.is_fitted = True
 529
 530        # Print final configuration
 531        print(f"\nFinal configuration:")
 532        print(f"  Best score: {self.best_score_:.6f}")
 533        print(f"  Best sigma: {sigma:.4f}")
 534        print(f"  Best lambda: {lambd:.6f}")
 535        if self.actual_use_rff_:
 536            print(
 537                f"  Using {self.kernel_approximation.upper()} with {self.actual_rff_components_} components"
 538            )
 539        else:
 540            print(f"  Using standard kernel method")
 541
 542        return self

 758    def _perm_test(self, Y_orig, Y_sim, stat_func, n_perm=1000):
 759        """Permutation test for distribution comparison."""
 760        if Y_orig.ndim == 1:
 761            Y_orig = Y_orig.reshape(-1, 1)
 762        if Y_sim.ndim == 1:
 763            Y_sim = Y_sim.reshape(-1, 1)
 764
 765        obs = stat_func(Y_orig, Y_sim)
 766        combined = np.vstack((Y_orig, Y_sim))
 767        n1 = Y_orig.shape[0]
 768        perms = np.zeros(n_perm)
 769
 770        for i in range(n_perm):
 771            idx = np.random.permutation(combined.shape[0])
 772            p1 = combined[idx[:n1]]
 773            p2 = combined[idx[n1:]]
 774            perms[i] = stat_func(p1, p2)
 775
 776        pval = (np.sum(perms >= obs) + 1) / (n_perm + 1)
 777        return obs, pval

 779    def _fisher_z_test(self, r1, r2, n1, n2):
 780        """Fisher z-test for comparing correlation coefficients."""
 781        z1 = np.arctanh(r1)
 782        z2 = np.arctanh(r2)
 783        z = (z1 - z2) / np.sqrt(1 / (n1 - 3) + 1 / (n2 - 3))
 784        p = 2 * (1 - stats.norm.cdf(np.abs(z)))
 785        return z, p

 145            X2_sq = jnp.sum(X2**2, axis=1)[jnp.newaxis, :]
 146            return X1_sq + X2_sq - 2 * X1 @ X2.T
 147
 148        @jit
 149        def cdist_jax(X1, X2):
 150            return vmap(
 151                lambda x: vmap(lambda y: jnp.sqrt(jnp.sum((x - y) ** 2)))(X2)
 152            )(X1)
 153
 154        self._pairwise_sq_dists_jax = pairwise_sq_dists_jax
 155        self._cdist_jax = cdist_jax

 174    def _create_model(self, gamma, alpha, use_rff=None):
 175        """Create the appropriate model based on RFF setting."""
 176        if use_rff is None:
 177            use_rff = self.actual_use_rff_
 178
 179        if use_rff:
 180            # Use kernel approximation with Ridge regression
 181            if self.rff_gamma is not None:
 182                effective_gamma = self.rff_gamma
 183            else:
 184                effective_gamma = gamma
 185
 186            # Determine number of components
 187            n_components = self.actual_rff_components_
 188
 189            if self.kernel_approximation == "rff":
 190                approximator = RBFSampler(
 191                    gamma=effective_gamma,
 192                    n_components=n_components,
 193                    random_state=self.random_state,
 194                )
 195            else:  # nystroem
 196                approximator = Nystroem(
 197                    kernel="rbf",
 198                    gamma=effective_gamma,
 199                    n_components=n_components,
 200                    random_state=self.random_state,
 201                )
 202
 203            # Create pipeline with scaling, approximation, and Ridge
 204            pipeline = Pipeline(
 205                [
 206                    ("scaler", StandardScaler()),
 207                    ("approx", approximator),
 208                    ("ridge", Ridge(alpha=alpha)),
 209                ]
 210            )
 211            return pipeline
 212        else:
 213            # Standard KernelRidge
 214            return KernelRidge(kernel=self.kernel, gamma=gamma, alpha=alpha)

 275    def _pairwise_sq_dists(self, X1, X2):
 276        """Compute pairwise squared Euclidean distances."""
 277        if self.backend in ["gpu", "tpu"] and JAX_AVAILABLE:
 278            X1_jax = jnp.array(X1)
 279            X2_jax = jnp.array(X2)
 280            result = self._pairwise_sq_dists_jax(X1_jax, X2_jax)
 281            return np.array(result)
 282        else:
 283            X1 = np.atleast_2d(X1)
 284            X2 = np.atleast_2d(X2)
 285            return (
 286                np.sum(X1**2, axis=1)[:, np.newaxis]
 287                + np.sum(X2**2, axis=1)[np.newaxis, :]
 288                - 2 * X1 @ X2.T
 289            )

 328    def _mmd(self, u, v, kernel_sigma=1):
 329        """Maximum Mean Discrepancy between two distributions."""
 330        if u.ndim == 1:
 331            u = u.reshape(-1, 1)
 332        if v.ndim == 1:
 333            v = v.reshape(-1, 1)
 334
 335        def kmat(A, B):
 336            return np.exp(
 337                -self._pairwise_sq_dists(A, B) / (2 * kernel_sigma**2)
 338            )
 339
 340        return (
 341            np.mean(kmat(u, u)) + np.mean(kmat(v, v)) - 2 * np.mean(kmat(u, v))
 342        )
 343
 344    def _custom_energy_distance(self, u, v):
 345        """Energy distance between two distributions."""
 346        if u.ndim == 1:
 347            u = u.reshape(-1, 1)
 348        if v.ndim == 1:
 349            v = v.reshape(-1, 1)
 350
 351        n, d = u.shape
 352        m = v.shape[0]
 353
 354        if self.backend in ["gpu", "tpu"] and JAX_AVAILABLE:
 355            # JAX implementation
 356            u_jax = jnp.array(u)
 357            v_jax = jnp.array(v)
 358
 359            dist_xx = self._cdist_jax(u_jax, u_jax)
 360            dist_yy = self._cdist_jax(v_jax, v_jax)
 361            dist_xy = self._cdist_jax(u_jax, v_jax)
 362
 363            term1 = 2 * jnp.sum(dist_xy) / (n * m)
 364            term2 = jnp.sum(dist_xx) / (n * n)
 365            term3 = jnp.sum(dist_yy) / (m * m)
 366
 367            return float(term1 - term2 - term3)
 368        else:
 369            # NumPy implementation
 370            dist_xx = cdist(u, u, metric="euclidean")
 371            dist_yy = cdist(v, v, metric="euclidean")
 372            dist_xy = cdist(u, v, metric="euclidean")
 373
 374            term1 = 2 * np.sum(dist_xy) / (n * m)
 375            term2 = np.sum(dist_xx) / (n * n)
 376            term3 = np.sum(dist_yy) / (m * m)
 377            return term1 - term2 - term3

 402    def fit(
 403        self, Y, X_dist, n_train=None, metric="energy", n_trials=50, **kwargs
 404    ):
 405        """
 406        Fit the data generator to match the distribution of Y.
 407
 408        Parameters:
 409        -----------
 410        Y : array-like, shape (n_samples, n_features)
 411            Target multivariate data to emulate
 412        X_dist : callable
 413            Function that takes n_samples and returns input data X
 414        n_train : int, default=None
 415            Number of training samples (default: n_samples // 2)
 416        metric : str, default='energy'
 417            Distance metric for optimization ('energy', 'mmd', or 'wasserstein')
 418        n_trials : int, default=50
 419            Number of Optuna optimization trials
 420        **kwargs : dict
 421            Additional arguments for Optuna optimization
 422
 423        Returns:
 424        --------
 425        self : object
 426            Returns self
 427        """
 428        if Y.ndim == 1:
 429            Y = Y.reshape(-1, 1)
 430
 431        n, d = Y.shape
 432        self.n_features_ = d
 433
 434        # Determine whether to use RFF
 435        if self.use_rff == "auto":
 436            self.actual_use_rff_ = n >= self.force_rff_threshold
 437        else:
 438            self.actual_use_rff_ = self.use_rff
 439
 440        # Auto-enable RFF for large datasets with component determination
 441        if self.actual_use_rff_:
 442            self.actual_rff_components_ = self._determine_components(n, d)
 443            if self.use_rff == "auto":
 444                print(
 445                    f"Large dataset detected (n={n}). Auto-enabling {self.kernel_approximation.upper()} for scalability."
 446                )
 447
 448        if n_train is None:
 449            n_train = n // 2
 450
 451        # Store the input distribution function
 452        self.X_dist = X_dist
 453
 454        # Create stratified train-test split
 455        train_idx, test_idx = self._stratified_train_test_split(Y, n_train)
 456        Y_train = Y[train_idx]
 457        Y_test = Y[test_idx]
 458        X_train = X_dist(n_train)
 459
 460        print(
 461            f"Stratified split: {len(train_idx)} train, {len(test_idx)} test samples"
 462        )
 463        print(f"Residual sampling method: {self.residual_sampling}")
 464        print(
 465            f"Using {self.kernel_approximation.upper()}: {self.actual_use_rff_}"
 466        )
 467        if self.actual_use_rff_:
 468            print(f"Components: {self.actual_rff_components_}")
 469            print(f"Approximation method: {self.kernel_approximation}")
 470        if self.residual_sampling == "kde":
 471            print(f"KDE bandwidth: {self.kde_bandwidth}")
 472        elif self.residual_sampling == "gmm":
 473            print(f"GMM components: {self.gmm_components}")
 474
 475        def objective(trial):
 476            sigma = trial.suggest_float("sigma", 0.01, 10, log=True)
 477            lambd = trial.suggest_float("lambd", 1e-5, 1, log=True)
 478            gamma = 1 / (2 * sigma**2)
 479
 480            # Create model with current parameters
 481            model = self._create_model(gamma, lambd)
 482            model.fit(X_train, Y_train)
 483
 484            preds_train = model.predict(X_train)
 485            if preds_train.ndim == 1:
 486                preds_train = preds_train.reshape(-1, 1)
 487
 488            res = Y_train - preds_train
 489            Y_sim = self._generate_pseudo_with_model(model, res, len(Y_test))
 490
 491            if metric == "energy":
 492                dist_val = self._custom_energy_distance(Y_test, Y_sim)
 493            elif metric == "mmd":
 494                dist_val = self._mmd(Y_test, Y_sim)
 495            elif metric == "wasserstein" and d == 1:
 496                dist_val = stats.wasserstein_distance(
 497                    Y_test.flatten(), Y_sim.flatten()
 498                )
 499            else:
 500                raise ValueError("Invalid metric for dimension")
 501
 502            return dist_val
 503
 504        # Optimize hyperparameters
 505        study = optuna.create_study(direction="minimize")
 506        study.optimize(objective, n_trials=n_trials, **kwargs)
 507
 508        # Store best parameters and fit final model
 509        self.best_params_ = study.best_params
 510        self.best_score_ = study.best_value
 511
 512        sigma = self.best_params_["sigma"]
 513        lambd = self.best_params_["lambd"]
 514        gamma = 1 / (2 * sigma**2)
 515
 516        # Fit final model with best parameters
 517        self.model = self._create_model(gamma, lambd)
 518        self.model.fit(X_train, Y_train)
 519
 520        # Compute residuals
 521        preds_train = self.model.predict(X_train)
 522        if preds_train.ndim == 1:
 523            preds_train = preds_train.reshape(-1, 1)
 524        self.residuals = Y_train - preds_train
 525
 526        # Fit the residual sampler
 527        self._fit_residual_sampler()
 528
 529        self.is_fitted = True
 530
 531        # Print final configuration
 532        print(f"\nFinal configuration:")
 533        print(f"  Best score: {self.best_score_:.6f}")
 534        print(f"  Best sigma: {sigma:.4f}")
 535        print(f"  Best lambda: {lambd:.6f}")
 536        if self.actual_use_rff_:
 537            print(
 538                f"  Using {self.kernel_approximation.upper()} with {self.actual_rff_components_} components"
 539            )
 540        else:
 541            print(f"  Using standard kernel method")
 542
 543        return self

 759    def _perm_test(self, Y_orig, Y_sim, stat_func, n_perm=1000):
 760        """Permutation test for distribution comparison."""
 761        if Y_orig.ndim == 1:
 762            Y_orig = Y_orig.reshape(-1, 1)
 763        if Y_sim.ndim == 1:
 764            Y_sim = Y_sim.reshape(-1, 1)
 765
 766        obs = stat_func(Y_orig, Y_sim)
 767        combined = np.vstack((Y_orig, Y_sim))
 768        n1 = Y_orig.shape[0]
 769        perms = np.zeros(n_perm)
 770
 771        for i in range(n_perm):
 772            idx = np.random.permutation(combined.shape[0])
 773            p1 = combined[idx[:n1]]
 774            p2 = combined[idx[n1:]]
 775            perms[i] = stat_func(p1, p2)
 776
 777        pval = (np.sum(perms >= obs) + 1) / (n_perm + 1)
 778        return obs, pval

 780    def _fisher_z_test(self, r1, r2, n1, n2):
 781        """Fisher z-test for comparing correlation coefficients."""
 782        z1 = np.arctanh(r1)
 783        z2 = np.arctanh(r2)
 784        z = (z1 - z2) / np.sqrt(1 / (n1 - 3) + 1 / (n2 - 3))
 785        p = 2 * (1 - stats.norm.cdf(np.abs(z)))
 786        return z, p

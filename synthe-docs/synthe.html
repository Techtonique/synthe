 85    def _oversample_uniform(self, chosen_bins):
 86        X_sampled = []
 87        for b in chosen_bins:
 88            multi_idx = np.unravel_index(b, (self.n_bins,) * self.d)
 89            coords = []
 90            for j, bi in enumerate(multi_idx):
 91                left = self.bin_edges[j][bi]
 92                right = self.bin_edges[j][bi + 1]
 93                coords.append(self.rng.uniform(left, right))
 94            X_sampled.append(coords)
 95        return np.array(X_sampled)

 281                + np.sum(X2**2, axis=1)[np.newaxis, :]

 314    def _mmd(self, u, v, kernel_sigma=1):
 315        """Maximum Mean Discrepancy between two distributions."""
 316        if u.ndim == 1:
 317            u = u.reshape(-1, 1)
 318        if v.ndim == 1:
 319            v = v.reshape(-1, 1)
 320
 321        def kmat(A, B):
 322            return np.exp(
 323                -self._pairwise_sq_dists(A, B) / (2 * kernel_sigma**2)
 324            )
 325
 326        return (
 327            np.mean(kmat(u, u)) + np.mean(kmat(v, v)) - 2 * np.mean(kmat(u, v))
 328        )

 330    def _custom_energy_distance(self, u, v):
 331        """Energy distance between two distributions."""
 332        if u.ndim == 1:
 333            u = u.reshape(-1, 1)
 334        if v.ndim == 1:
 335            v = v.reshape(-1, 1)
 336
 337        n, d = u.shape
 338        m = v.shape[0]
 339
 340        if self.backend in ["gpu", "tpu"] and JAX_AVAILABLE:
 341            # JAX implementation
 342            u_jax = jnp.array(u)
 343            v_jax = jnp.array(v)
 344            dist_xx = self._cdist_jax(u_jax, u_jax)
 345            dist_yy = self._cdist_jax(v_jax, v_jax)
 346            dist_xy = self._cdist_jax(u_jax, v_jax)
 347            term1 = 2 * jnp.sum(dist_xy) / (n * m)
 348            term2 = jnp.sum(dist_xx) / (n * n)
 349            term3 = jnp.sum(dist_yy) / (m * m)
 350            return float(term1 - term2 - term3)
 351        else:
 352            # NumPy implementation
 353            dist_xx = cdist(u, u, metric="euclidean")
 354            dist_yy = cdist(v, v, metric="euclidean")
 355            dist_xy = cdist(u, v, metric="euclidean")
 356            term1 = 2 * np.sum(dist_xy) / (n * m)
 357            term2 = np.sum(dist_xx) / (n * n)
 358            term3 = np.sum(dist_yy) / (m * m)
 359            return term1 - term2 - term3

 770    def _perm_test(self, Y_orig, Y_sim, stat_func, n_perm=1000):
 771        """Permutation test for distribution comparison."""
 772        if Y_orig.ndim == 1:
 773            Y_orig = Y_orig.reshape(-1, 1)
 774        if Y_sim.ndim == 1:
 775            Y_sim = Y_sim.reshape(-1, 1)
 776
 777        obs = stat_func(Y_orig, Y_sim)
 778        combined = np.vstack((Y_orig, Y_sim))
 779        n1 = Y_orig.shape[0]
 780        perms = np.zeros(n_perm)
 781
 782        for i in range(n_perm):
 783            idx = np.random.permutation(combined.shape[0])
 784            p1 = combined[idx[:n1]]
 785            p2 = combined[idx[n1:]]
 786            perms[i] = stat_func(p1, p2)
 787
 788        pval = (np.sum(perms >= obs) + 1) / (n_perm + 1)
 789        return obs, pval

 791    def _fisher_z_test(self, r1, r2, n1, n2):
 792        """Fisher z-test for comparing correlation coefficients."""
 793        z1 = np.arctanh(r1)
 794        z2 = np.arctanh(r2)
 795        z = (z1 - z2) / np.sqrt(1 / (n1 - 3) + 1 / (n2 - 3))
 796        p = 2 * (1 - stats.norm.cdf(np.abs(z)))
 797        return z, p

 799    def test_similarity(self, Y_orig, Y_sim, n_perm=1000):
 800        """
 801        Test statistical similarity between original and synthetic data.
 802
 803        Parameters:
 804        -----------
 805        Y_orig : array-like
 806            Original data
 807        Y_sim : array-like
 808            Synthetic data
 809        n_perm : int, default=1000
 810            Number of permutations for permutation tests
 811
 812        Returns:
 813        --------
 814        results : dict
 815            Dictionary containing test results
 816        """
 817        if Y_orig.ndim == 1:
 818            Y_orig = Y_orig.reshape(-1, 1)
 819        if Y_sim.ndim == 1:
 820            Y_sim = Y_sim.reshape(-1, 1)
 821
 822        d = Y_orig.shape[1]
 823        results = {}
 824        # Test 1: Perm with energy
 825        results["energy_perm"] = self._perm_test(
 826            Y_orig, Y_sim, self._custom_energy_distance, n_perm
 827        )
 828        # Test 2: Perm with MMD
 829        results["mmd_perm"] = self._perm_test(
 830            Y_orig, Y_sim, lambda u, v: self._mmd(u, v), n_perm
 831        )
 832
 833        # Test 3: Perm with avg Wasserstein on margins
 834        def avg_wass(u, v):
 835            return np.mean(
 836                [stats.wasserstein_distance(u[:, i], v[:, i]) for i in range(d)]
 837            )
 838
 839        results["avg_wass_perm"] = self._perm_test(
 840            Y_orig, Y_sim, avg_wass, n_perm
 841        )
 842        # Test 4: Min p-value from marginal KS tests
 843        ps_ks = [
 844            stats.ks_2samp(Y_orig[:, i], Y_sim[:, i]).pvalue for i in range(d)
 845        ]
 846        results["min_marginal_ks_p"] = min(ps_ks)

 849            stats.anderson_ksamp([Y_orig[:, i], Y_sim[:, i]]).significance_level
